100    pub fn tx_event_key_parse(&self, send_event: &str) -> Result<(u64, String), String> {
101        let prefix = self.tx_event_key_prefix();
102        if !send_event.starts_with(&prefix) {
103            return Err(format!(
104                "Invalid send_event format: missing prefix '{}': {}",
105                prefix, send_event
106            ));
107        }
108        let rest = &send_event[prefix.len()..];
109        let mut parts = rest.splitn(2, ':');
110        let tick_str = parts.next().ok_or_else(|| "Missing tick".to_string())?;
111        let msg_id = parts
112            .next()
113            .ok_or_else(|| "Missing msg_id".to_string())?
114            .to_string();
115        let send_at_tick = tick_str
116            .parse::<u64>()
117            .map_err(|e| format!("failed to parse tick: {}", e))?;
118        Ok((send_at_tick, msg_id))
119    }

217pub struct Courier {
218    /// The core message delivery state machine for reliable, persistent, and retryable messaging.
219    ///
220    /// # Construction
221    /// To construct a [`Courier`], provide the following:
222    ///
223    /// - `db`: An implementation of [`courier::db::DB`] for message persistence.
224    /// - `sender`: An implementation of [`Tx`] to perform outbound message delivery.
225    /// - `handler`: A function or closure to handle inbound messages after receipt.
226    /// - `pact_factory`: A closure that produces an initial [`Pact`] for each message during [`Courier::tx`].
227    /// - `pact_ticker`: A closure that determines the retry schedule. It is called on each send attempt and returns `Ok(next_tick)` for the next retry, or `Err(reason)` to stop retrying. [`Courier`] will update `Pact.tick_of_last_attempt` automatically.
228    ///
229    /// This design enables flexible integration with custom storage, retry, and communication strategies.
230    id: String,
231    db: Arc<dyn DB + Send + Sync>,
232    dao: DAO,
233    pact_factory: Arc<dyn Fn(&Msg) -> Pact + Send + Sync>,
234    pact_ticker: Arc<dyn Fn(&mut Pact, u64) -> Option<u64> + Send + Sync>, // New pact_ticker field
235    keep_receipt: Arc<dyn Fn(&Receipt) -> bool + Send + Sync>,
236    sender: Arc<dyn Tx + Send + Sync>,
237    tick_last: AtomicU64,
238}

269    /// Advances the Courier state machine for a given logical tick.
270    ///
271    /// This method processes all scheduled send events whose tick is due, attempts message delivery,
272    /// manages retry scheduling, and expires receipts as needed. It is intended to be called periodically
273    /// (e.g., by a timer or event loop) to drive message delivery and cleanup.
274    ///
275    /// # Arguments
276    /// * `ctx` - Shared context for the operation (used for async message sending).
277    /// * `dbtx` - An open database transaction for all reads and writes.
278    /// * `tick` - The current logical tick (time step) for scheduling.
279    ///
280    /// # Returns
281    /// * `Ok(())` on success, or an error string if any operation fails.
282    ///
283    /// # Transaction Management
284    /// All database operations are performed within the provided transaction. The caller is responsible
285    /// for committing or rolling back the transaction as appropriate.
286    pub async fn tick(
287        &self,
288        ctx: Arc<RwLock<Context>>,
289        dbtx: &dyn DBTx,
290        tick: u64,
291    ) -> Result<(), String> {
292        info!("Tick started.");
293
294        // handle send events
295        let tx_event_keys = dbtx
296            .seq_get(&self.dao.tx_event_key_prefix())
297            .inspect_err(|e| {
298                error!("failed to retrieve tick sequence: {}", e);
299            })?;
300        for tx_event_key in tx_event_keys {
301            info!("Scanning key: {}", tx_event_key); // Debugging log
302
303            // parse the tx_event_key
304            let (send_tick, msg_id) = self.dao.tx_event_key_parse(&tx_event_key)?;
305
306            // if send_tick > tick, the send_event is in the future
307            if send_tick > tick {
308                // we assume the scan happens in order, so we are done ticking
309                break;
310            }
311
312            // delete the send_event from the tick sequence
313            self.dao.tx_event_del(dbtx, &msg_id, send_tick)?;
314
315            // retrieve the msg from the db
316            let msg = match self.dao.tx_msg_get(dbtx, &msg_id)? {
317                Some(msg) => msg,
318                None => {
319                    // should never happen...
320                    // but what if our db allows a delete within this dbtx?
321                    error!("msg_id {} deleted while processing send_event", msg_id);
322                    continue;
323                }
324            };
325
326            // get the pact from the db
327            let mut pact: Pact = match self.dao.tx_pact_get(dbtx, &msg_id)? {
328                Some(pact) => pact,
329                None => {
330                    // should never happen...
331                    // but what if our db allows a delete within this dbtx?
332                    error!(
333                        "pact for msg_id {} deleted while processing send_event",
334                        msg_id
335                    );
336                    continue;
337                }
338            };
339
340            // should we schedule a next send_event?
341            if let Some(next_send_tick) = (self.pact_ticker)(&mut pact, tick) {
342                // yes. save a new send event
343                self.dao.tx_event_put(dbtx, &msg_id, next_send_tick)?;
344
345                // update and save the pact
346                pact.tick_of_last_attempt = tick;
347                self.dao.tx_pact_put(dbtx, &msg_id, &pact)?;
348            } else {
349                // no more retries
350                self.dao.tx_msg_del(dbtx, &msg.id)?;
351                self.dao.tx_pact_del(dbtx, &msg.id)?;
352            }
353
354            // send and await
355            if let Err(e) = self.sender.tx(ctx.clone(), &msg).await {
356                error!("failed to send message: {}", e);
357            }
358        }
359
360        // now expire receipts
361        let rx_receipt_keys = dbtx
362            .seq_get(&self.dao.rx_receipt_key_prefix())
363            .map_err(|e| format!("failed to retrieve tick sequence: {}", e))?;
364
365        for rx_receipt_key in rx_receipt_keys {
366            let msg_id = self.dao.rx_receipt_key_parse(&rx_receipt_key);
367            let mut rx_receipt = match self.dao.rx_receipt_get(dbtx, &msg_id)? {
368                Some(r) => r,
369                None => {
370                    // should never happen...
371                    // but what if our db allows a delete within this dbtx?
372                    error!(
373                        "receipt disappeared while iterating in tick {}",
374                        rx_receipt_key
375                    );
376                    continue;
377                }
378            };
379
380            // let keep_receipt decide
381            if (self.keep_receipt)(&rx_receipt) {
382                rx_receipt.tick_acked_last = tick;
383                self.dao.rx_receipt_put(dbtx, &msg_id, &rx_receipt)?;
384            } else {
385                self.dao.rx_receipt_del(dbtx, &msg_id)?;
386            }
387        }
388
389        Ok(())
390    }
391}

393#[async_trait]
394impl Rx for Courier {
395    /// Handles inbound messages and manages their persistence and acknowledgment.
396    ///
397    /// This method is called when a message is received. It creates and commits a new database transaction
398    /// for each inbound message. If the message is an ACK, it deletes the corresponding outbound message and pact.
399    /// For standard messages, it stores the message and a receipt, or updates the receipt if already present.
400    /// An ACK is always sent in response to a standard message.
401    ///
402    /// # Arguments
403    /// * `ctx` - Shared context for the operation (used for async message sending).
404    /// * `msg` - The inbound message to process.
405    ///
406    /// # Returns
407    /// * `Ok(())` on success, or an error string if any operation fails.
408    ///
409    /// # Transaction Management
410    /// A new transaction is created and committed for each call.
411    async fn rx(&self, ctx: Arc<RwLock<Context>>, msg: &Msg) -> Result<(), String> {
412        print!("rx");
413
414        let dbtx_arc = self.db.dbtx_create().map_err(|e| e.to_string())?;
415        let dbtx = dbtx_arc.as_ref();
416
417        // is the incoming message an ack?
418        if msg.type_ == *MSG_ACK_TYPE {
419            self.dao.tx_msg_del(dbtx, &msg.body)?;
420            self.dao.tx_pact_del(dbtx, &msg.body)?;
421        } else {
422            // do we have a receipt for the inbound message?
423            let rx_receipt_key = self.dao.rx_receipt_key_make(&msg.id);
424            if let Some(mut tx_receipt) = self.dao.rx_receipt_get(dbtx, &rx_receipt_key)? {
425                // yes. update receipt.tick_acked_last and re-save
426                tx_receipt.tick_acked_last = self.tick_last.load(Relaxed);
427                self.dao.rx_receipt_put(dbtx, &msg.id, &tx_receipt)?;
428                info!("Receipt updated for key: {}", rx_receipt_key);
429            } else {
430                // save the inbound message for pickup later
431                self.dao.rx_msg_put(dbtx, &msg)?;
432
433                // save a receipt
434                let receipt = Receipt {
435                    tick_acked_first: self.tick_last.load(Relaxed),
436                    tick_acked_last: self.tick_last.load(Relaxed),
437                };
438                self.dao.rx_receipt_put(dbtx, &msg.id, &receipt)?;
439
440                info!(
441                    "Message and receipt successfully stored for key: {}",
442                    &msg.id
443                );
444            }
445        }
446
447        // commit the tx (before sending the ack)
448        dbtx.commit().map_err(|e| {
449            error!("failed to commit rx transaction {}", e);
450            e
451        })?;
452
453        if msg.type_ != *MSG_ACK_TYPE {
454            // send an ack
455            let ack_msg = Msg {
456                id: uuid::Uuid::new_v4().to_string(),
457                from_id: self.id.clone(),
458                to_ids: vec![msg.from_id.clone()],
459                type_: MSG_ACK_TYPE.clone(),
460                body: msg.id.clone(),
461            };
462
463            // await the send and log errors
464            let send_result = self.sender.tx(ctx, &ack_msg).await;
465            if let Err(e) = send_result {
466                error!("failed to send ack: {}", e);
467            }
468        }
469
470        Ok(())
471    }
472}

474#[async_trait]
475impl Tx for Courier {
476    /// Schedules a message for delivery and persists its state.
477    ///
478    /// This method is called to send a message. It writes the message and its initial pact to the database,
479    /// and schedules an event for immediate delivery. All operations are performed within the provided transaction.
480    /// The transaction is not committed here; the caller is responsible for committing or rolling back as needed.
481    ///
482    /// # Arguments
483    /// * `ctx` - Shared context for the operation (must contain a valid database transaction).
484    /// * `msg` - The message to send.
485    ///
486    /// # Returns
487    /// * `Ok(())` on success, or an error string if any operation fails.
488    ///
489    /// # Transaction Management
490    /// All database operations are performed within the transaction from the context.
491    async fn tx(&self, ctx: Arc<RwLock<Context>>, msg: &Msg) -> Result<(), String> {
492        // Retrieve dbtx from context
493        let ctx_guard = ctx.read().unwrap();
494        let dbtx = dbtx_from_ctx(&ctx_guard).unwrap();
495        let dbtx = dbtx.as_ref();
496
497        // save the message to the db
498        self.dao.tx_msg_put(dbtx, &msg)?;
499
500        // save a pact
501        let pact = (self.pact_factory)(msg);
502        self.dao.tx_pact_put(dbtx, &msg.id, &pact)?;
503
504        // schedule an event to send asap
505        self.dao.tx_event_put(dbtx, &msg.id, 0)?;
506
507        Ok(())
508    }
509}
